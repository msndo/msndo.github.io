easeOutExpoQuick(Custom)

$.easing["easeOutExpoQuick"] = function(x, t, b, c, d){ // (x:進捗率 0 - 1, t:経過時間msec, b:0固定, c:1固定, d:トータル時間msec)
	return (t==d) ? b+c : c * (-Math.pow(10, -15 * t/d) + 1) + b;
};

easeInOutExpo(Preset)

easeInOutExpo: function (x, t, b, c, d) {
	if (t==0) return b;
	if (t==d) return b+c;
	if ((t/=d/2) < 1) return c/2 * Math.pow(2, 10 * (t - 1)) + b;
	return c/2 * (-Math.pow(2, -10 * --t) + 2) + b;
}
Start

easeInOutExpoLongTail(Custom)

$.easing["easeInOutExpoLongTail"] = function(x, t, b, c, d){ // (x:進捗率 0 - 1, t:経過時間msec, b:0固定, c:1固定, d:トータル時間msec)
	if (t==0) return b;
	if (t==d) return b+c;

	if ((t/=d/2) < 1) return c/2 * Math.pow(8, 10 * (t - 1)) + b;
	return c/2 * (-Math.pow(8, -3 * --t) + 2) + b;
};

easeTimeline(Experiment)

$.easing["easeTimeline"] = function(x, t, b, c, d){ // (x:進捗率 0 - 1, t:経過時間msec, b:0固定, c:1固定, d:トータル時間msec)
	var diviation = 1;
	if(x < 0.05) {
		diviation = 0.15;
	}
	else if(x < 0.1) {
		diviation = 0.2;
	}
	else if(x < 0.2) {
		diviation = 0.17;
	}
	else if(x < 0.3) {
		diviation = 0.2;
	}
	else if(x < 0.4) {
		diviation = 0.1;
	}
	else if(x < 0.5) {
		diviation = 0.4;
	}
	else if(x < 0.6) {
		diviation = 0.7;
	}
	else if(x < 0.7) {
		diviation = 0.9;
	}
	else if(x < 0.8) {

	}
	else if(x < 0.9) {
